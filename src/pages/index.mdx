```js
function Navlink({
  text = "Navlink",
  to = "/components",
  // Mocked reach router location object
  location = { pathname: "/" },
}) {
  return (
    <Link
      className="nav-link"
      data-is-active={location.pathname === to}
      to={to}
    >
      {text}
    </Link>
  )
}
```
